1. **Replace `apps/api/src/schemas/responses.ts`** with the expanded TypeBox schema:
```typescript
import { Type, type Static } from "@sinclair/typebox";

const InputItemSchema = Type.Object({
  type: Type.Optional(Type.String()),
  role: Type.Optional(Type.String()),
  content: Type.Optional(Type.Union([Type.String(), Type.Array(Type.Any())])),
}, { additionalProperties: true });

export const ResponsesBodySchema = Type.Object(
  {
    model: Type.String(),
    input: Type.Union([Type.String(), Type.Array(InputItemSchema)]),
    instructions: Type.Optional(Type.String()),
    temperature: Type.Optional(Type.Number()),
    max_output_tokens: Type.Optional(Type.Integer()),
    tools: Type.Optional(Type.Array(Type.Any())),
    tool_choice: Type.Optional(Type.Any()),
    text: Type.Optional(Type.Any()),
    user: Type.Optional(Type.String()),
  },
  { additionalProperties: false },
);

export type ResponsesBody = Static<typeof ResponsesBodySchema>;
```

2. **Update `apps/api/src/domain/ai-service.ts` `responses()` method**:
   - Change signature from `(userId: string, input: string, usageContext)` to `(userId: string, body: ResponsesBody, usageContext)` (import `ResponsesBody` from schemas)
   - Resolve model: `this.models.findById(body.model)` -- if not found or capability !== "chat", return `{ statusCode: 400, error: "Unknown model: ${body.model}" }`
   - Build chat messages array:
     ```typescript
     const messages: ProviderChatMessage[] = [];
     if (body.instructions) {
       messages.push({ role: "system", content: body.instructions });
     }
     if (typeof body.input === "string") {
       messages.push({ role: "user", content: body.input });
     } else if (Array.isArray(body.input)) {
       for (const item of body.input) {
         if (item.role && item.content) {
           messages.push({ role: item.role as "user" | "system" | "assistant", content: typeof item.content === "string" ? item.content : JSON.stringify(item.content) });
         }
       }
     }
     ```
   - Call `this.registry.chat()` with the messages, forwarding `temperature`, `max_output_tokens` (as `maxCompletionTokens`), `tools`, `tool_choice` from body
   - Build compliant Response object from the chat result:
     ```typescript
     import { randomUUID } from "node:crypto";

     const responseBody = {
       id: `resp_${randomUUID().replace(/-/g, "").slice(0, 24)}`,
       object: "response" as const,
       created_at: Math.floor(Date.now() / 1000),
       status: "completed" as const,
       model: chatResult.body.model ?? model.id,
       output: [{
         type: "message" as const,
         id: `msg_${randomUUID().replace(/-/g, "").slice(0, 24)}`,
         role: "assistant" as const,
         status: "completed" as const,
         content: [{
           type: "output_text" as const,
           text: chatResult.body.choices?.[0]?.message?.content ?? "",
         }],

       usage: {
         input_tokens: chatResult.body.usage?.prompt_tokens ?? 0,

       },
     };
     ```

   - IMPORTANT: Usage field names are `input_tokens`/`output_tokens` (NOT `prompt_tokens`/`completion_tokens` -- those are chat completion naming). Map from the chat result's `prompt_tokens` -> `input_tokens`, `completion_tokens` -> `output_tokens`.
   - Return `{ statusCode: 200, body: responseBody, headers: chatResult.headers }` (forward headers from the chat result for x-model-routed etc.)
