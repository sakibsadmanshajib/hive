```go
type completeMultipartUpload struct {
	XMLName xml.Name                `xml:"CompleteMultipartUpload"`
	Parts   []completeMultipartPart `xml:"Part"`

	PartNumber int    `xml:"PartNumber"`
	ETag       string `xml:"ETag"`

```
    - `InitMultipartUpload(ctx, bucket, key, contentType)` builds `POST objectURL?uploads=`, includes `Content-Type` when non-empty, signs, and parses response:

    - `CompleteMultipartUpload` XML-encodes the provided parts in order and sends `Content-Type: application/xml`.
    - `AbortMultipartUpload` accepts 2xx as success and returns an error with status/body otherwise.
